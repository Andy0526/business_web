{
    平台名称:{
        "reviews"{
            "0"(不推荐):{
                "labels":{
                    "平台大":5(出现次数),
                    "提现快":6
                    ...
                }
                "comments":[
                    {
                        "content":评论内容,
                        "date":评论时间
                    }
                ]
            }
            "1"(一般):{
                "labels":{},
                "comments":{}
            }
            "2"(推荐):{
                "labels":{},
                "comments":{}
            }
        }
    }

}
